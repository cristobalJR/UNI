SELECT *
FROM DEPARTAMENTO;


--1.Nombre y apellidos de empleados que son jefes
-- la identificacion de los jefes

    SELECT NOMBRE, APELLIDO
    FROM EMPLEADO
    WHERE ID_EMPLEADO IN (SELECT DISTINCT (JEFE)
                            FROM EMPLEADO 
                            WHERE JEFE IS NOT NULL);
        
--2. NOMBRE Y APELLIDOS DE LOS EMPLEADOS QUE NO TIENEN SALARIO

    SELECT NOMBRE, APELLIDO
    FROM EMPLEADO
    WHERE SALARIO IS NULL
            OR SALARIO = 0;
            
--OTRA FORMA
    
    SELECT NOMBRE, APELLIDO
    FROM EMPLEADO
    WHERE NVL(SALARIO,0)=0;
    
--3 NOMBRE Y APELLIDO Y ANTIGUEDAD DE LOS EMPLEADOS QUE EN SU APELLIDO CONTIENEN EL TEXTO 'AE' O TERMINA EN 'I'

    SELECT NOMBRE, APELLIDO, ANTIGUEDAD
    FROM EMPLEADO
    WHERE APELLIDO LIKE '%ae%' OR APELLIDO LIKE '%i';
    

--4.NUMERO DE EMPLEADPOS EN MI BASE DE DATOS

    SELECT COUNT(ID_EMPLEADO)
    FROM EMPLEADO;
    
--5. SALARIO MEDIO DE LOS EMPLEADOS

    SELECT AVG(SALARIO)
    FROM EMPLEADO;

--6 NUMERO DE EMPLEADOS QUE COMBRAR MAS DE 70000 EUROS

    SELECT COUNT(*)
    FROM EMPLEADO
    WHERE SALARIO > 70000;
    
--7SALARIO MAS ALTO, SALARIO MAS BAJO Y SALARIO PROMEDIO DEL DEPARTAMENTO NUMERO 10

    SELECT MAX(SALARIO), MIN(SALARIO), AVG(SALARIO)
        FROM EMPLEADO
        WHERE ID_DEPARTAMENTO=10;
    
--8 OBTENER POR CADA DEPARTAMENTO EL NUMERO DE EMPLEADOS Y SALARIO  MEDIO

    SELECT ID_DEPARTAMENTO, COUNT(ID_EMPLEADO), AVG(SALARIO)
        FROM EMPLEADO
        GROUP BY ID_DEPARTAMENTO;
        
--9obtener de cada departamento el numero deempleados y salario medio de aquellos que tengan un jefe asignado

    SELECT ID_DEPARTAMENTO, COUNT(ID_EMPLEADO), AVG(SALARIO)
        FROM EMPLEADO
        WHERE JEFE IS NOT NULL
        GROUP BY ID_DEPARTAMENTO;
        
--10 OBTENER POR CADA DEPARTAMENTO QUE TENGA MAS DE UN EMPLEADO
-- EL NUMERO DE EMPLEADOS Y SALARIO MEDIO
-- DE AQUELLOS QUE TENGAN JEFE ASIGNADO

    SELECT ID_DEPARTAMENTO, COUNT(ID_EMPLEADO), AVG(SALARIO)
        FROM EMPLEADO
        WHERE JEFE IS NOT NULL
        GROUP BY ID_DEPARTAMENTO
        HAVING COUNT(ID_EMPLEADO)>1;
        
--11 ID DE EMPLEADO, AÑO DE INCORPORACION Y SALARIO MENSUAL

    SELECT ID_EMPLEADO, TO_CHAR(ANTIGUEDAD, 'yyyy')"ANTIGÜEDAD", ROUND(SALARIO/12, 2)"SALARIO_MENSUAL"
        FROM EMPLEADO;
        
--12 NUMERO DE EMPLEADOS Y SALARIO MAXIMO POR AÑO DE INCORPORACION

    SELECT TO_CHAR(ANTIGUEDAD, 'yyyy')"AÑO_INCORPORACION", COUNT(ID_EMPLEADO), MAX(NVL(SALARIO,0)), ROUND(AVG(NVL(SALARIO,0)), 2)
        FROM EMPLEADO
        GROUP BY TO_CHAR(ANTIGUEDAD, 'yyyy')
        ORDER BY TO_CHAR(ANTIGUEDAD, 'yyyy') ASC;
    
--13 NUMERO DE EMPLEADOS Y SALARIO MAXIMO POR AÑO DE INCORPORACION, CON MAS DE UN EMPLEADO

    SELECT TO_CHAR(ANTIGUEDAD, 'yyyy')"AÑO_INCORPORACION", COUNT(ID_EMPLEADO), MAX(NVL(SALARIO,0)), ROUND(AVG(NVL(SALARIO,0)), 2)
        FROM EMPLEADO
        GROUP BY TO_CHAR(ANTIGUEDAD, 'yyyy')
        HAVING COUNT(ID_EMPLEADO)>1
        ORDER BY TO_CHAR(ANTIGUEDAD, 'yyyy') ASC;

--16 AQUELLOS EMPLEADOS QUE NO TRABAJEN EN NINGUN PROYECTO
    SELECT ID_EMPLEADO FROM EMPLEADO
             MINUS
            SELECT ID_EMPLEADO FROM TRABAJA;
    --OTRA FORMA
    SELECT ID_EMPLEADO FROM EMPLEADO
    WHERE ID_EMPLEADO NOT IN(SELECT ID_EMPLEADO FROM TRABAJA);
