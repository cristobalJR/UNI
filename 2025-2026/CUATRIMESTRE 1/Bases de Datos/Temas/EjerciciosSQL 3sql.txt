-- 2. Nombre, apellidos y nombre del proyecto donde trabajan los empleados

SELECT E.NOMBRE, E.APELLIDO, P.NOMBRE "PROYECTO"
FROM EMPLEADO E
JOIN TRABAJA T  ON E.ID_EMPLEADO = T.ID_EMPLEADO
JOIN PROYECTO P ON P.ID_PROYECTO = T.ID_PROYECTO;

-- 3. Salario medio de los empleados por dirección

SELECT D.DIRECCION, AVG(E.SALARIO) "SALARIO_MEDIO"
FROM  EMPLEADO E
 JOIN DEPARTAMENTO D ON E.ID_DEPARTAMENTO = D.ID_DEPARTAMENTO
GROUP BY D.DIRECCION;


-- 4. Nombre del proyecto, nombre del empleado y fecha de comienzo de participación 
--   ordenado de forma descendente por fecha de incorporación

SELECT P.NOMBRE, E.NOMBRE, T.FECHA
FROM EMPLEADO E
JOIN TRABAJA T  ON E.ID_EMPLEADO = T.ID_EMPLEADO
JOIN PROYECTO P ON P.ID_PROYECTO = T.ID_PROYECTO
ORDER BY 3 DESC;

-- Crear una vista Empleados por Departamento que contenga el nombre de 
--  los departamentos y el número de empelados de cada uno de ellos.

CREATE OR REPLACE VIEW EMPLEADOS_POR_DEPARTAMENTO (NOMBRE_DEPARTAMENTO, NUM_EMPLEADOS) AS (
    SELECT D.NOMBRE, COUNT(ID_EMPLEADO)
    FROM DEPARTAMENTO D
    JOIN EMPLEADO E ON E.ID_DEPARTAMENTO = D.ID_DEPARTAMENTO
    GROUP BY D.NOMBRE);
